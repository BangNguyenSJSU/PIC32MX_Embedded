53void static UART2_ErrorClear( void )
54{
55 UART_ERROR errors = UART_ERROR_NONE;
56 uint8_t dummyData = 0u;
57
58 errors = (UART_ERROR)(U2STA & (_U2STA_OERR_MASK | _U2STA_FERR_MASK | _U2STA_PERR_MASK));
59
60 if(errors != UART_ERROR_NONE)
61 {
62 /* If it's a overrun error then clear it to flush FIFO */
63 if((U2STA & _U2STA_OERR_MASK) != 0U)
64 {
65 U2STACLR = _U2STA_OERR_MASK;
66 }
67
68 /* Read existing error bytes from FIFO to clear parity and framing error flags */
69 while((U2STA & _U2STA_URXDA_MASK) != 0U)
70 {
71 dummyData = (uint8_t)U2RXREG;
72 }
73
74 /* Clear error interrupt flag */
75 IFS1CLR = _IFS1_U2EIF_MASK;
76
77 /* Clear up the receive interrupt flag so that RX interrupt is not
78 * triggered for error bytes */
79 IFS1CLR = _IFS1_U2RXIF_MASK;
80 }
81
82 // Ignore the warning
83 (void)dummyData;
84}

86void UART2_Initialize( void )
87{
88 /* Set up UxMODE bits */
89 /* STSEL = 0 */
90 /* PDSEL = 0 */
91 /* UEN = 0 */
92
93 U2MODE = 0x8;
94
95 /* Enable UART2 Receiver and Transmitter */
96 U2STASET = (_U2STA_UTXEN_MASK | _U2STA_URXEN_MASK | _U2STA_UTXISEL1_MASK );
97
98 /* BAUD Rate register Setup */
99 U2BRG = 520;
100
101 /* Disable Interrupts */
102 IEC1CLR = _IEC1_U2EIE_MASK;
103
104 IEC1CLR = _IEC1_U2RXIE_MASK;
105
106 IEC1CLR = _IEC1_U2TXIE_MASK;
107
108 /* Initialize instance object */
109 uart2Obj.rxBuffer = NULL;
110 uart2Obj.rxSize = 0;
111 uart2Obj.rxProcessedSize = 0;
112 uart2Obj.rxBusyStatus = false;
113 uart2Obj.rxCallback = NULL;
114 uart2Obj.txBuffer = NULL;
115 uart2Obj.txSize = 0;
116 uart2Obj.txProcessedSize = 0;
117 uart2Obj.txBusyStatus = false;
118 uart2Obj.txCallback = NULL;
119 uart2Obj.errors = UART_ERROR_NONE;
120
121 /* Turn ON UART2 */
122 U2MODESET = _U2MODE_ON_MASK;
123}

125bool UART2_SerialSetup( UART_SERIAL_SETUP *setup, uint32_t srcClkFreq )
126{
127 bool status = false;
128 uint32_t baud;
129 uint32_t status_ctrl;
130 uint32_t uxbrg = 0;
131
132 if(uart2Obj.rxBusyStatus == true)
133 {
134 /* Transaction is in progress, so return without updating settings */
135 return status;
136 }
137 if (uart2Obj.txBusyStatus == true)
138 {
139 /* Transaction is in progress, so return without updating settings */
140 return status;
141 }
142
143 if (setup != NULL)
144 {
145 baud = setup->baudRate;
146
147 if ((baud == 0U) || ((setup->dataWidth == UART_DATA_9_BIT) && (setup->parity != UART_PARITY_NONE)))
148 {
149 return status;
150 }
151
152 if(srcClkFreq == 0U)
153 {
154 srcClkFreq = UART2_FrequencyGet();
155 }
156
157 /* Calculate BRG value */
158 uxbrg = (((srcClkFreq >> 2) + (baud >> 1)) / baud);
159 /* Check if the baud value can be set with low baud settings */
160 if (uxbrg < 1U)
161 {
162 return status;
163 }
164
165 uxbrg -= 1U;
166
167 if (uxbrg > UINT16_MAX)
168 {
169 return status;
170 }
171
172 /* Turn OFF UART2. Save UTXEN, URXEN and UTXBRK bits as these are cleared upon disabling UART */
173
174 status_ctrl = U2STA & (_U2STA_UTXEN_MASK | _U2STA_URXEN_MASK | _U2STA_UTXBRK_MASK);
175
176 U2MODECLR = _U2MODE_ON_MASK;
177
178 if(setup->dataWidth == UART_DATA_9_BIT)
179 {
180 /* Configure UART2 mode */
181 U2MODE = (U2MODE & (~_U2MODE_PDSEL_MASK)) | setup->dataWidth;
182 }
183 else
184 {
185 /* Configure UART2 mode */
186 U2MODE = (U2MODE & (~_U2MODE_PDSEL_MASK)) | setup->parity;
187 }
188
189 /* Configure UART2 mode */
190 U2MODE = (U2MODE & (~_U2MODE_STSEL_MASK)) | setup->stopBits;
191
192 /* Configure UART2 Baud Rate */
193 U2BRG = uxbrg;
194
195 U2MODESET = _U2MODE_ON_MASK;
196
197 /* Restore UTXEN, URXEN and UTXBRK bits. */
198 U2STASET = status_ctrl;
199
200 status = true;
201 }
202
203 return status;
204}

228bool UART2_Read(void* buffer, const size_t size )
229{
230 bool status = false;
231
232 if(buffer != NULL)
233 {
234 /* Check if receive request is in progress */
235 if(uart2Obj.rxBusyStatus == false)
236 {
237 /* Clear error flags and flush out error data that may have been received when no active request was pending */
238 UART2_ErrorClear();
239
240 uart2Obj.rxBuffer = buffer;
241 uart2Obj.rxSize = size;
242 uart2Obj.rxProcessedSize = 0;
243 uart2Obj.rxBusyStatus = true;
244 uart2Obj.errors = UART_ERROR_NONE;
245 status = true;
246
247 /* Enable UART2_FAULT Interrupt */
248 IEC1SET = _IEC1_U2EIE_MASK;
249
250 /* Enable UART2_RX Interrupt */
251 IEC1SET = _IEC1_U2RXIE_MASK;
252 }
253 }
254
255 return status;
256}

258bool UART2_Write( void* buffer, const size_t size )
259{
260 bool status = false;
261
262 if(buffer != NULL)
263 {
264 /* Check if transmit request is in progress */
265 if(uart2Obj.txBusyStatus == false)
266 {
267 uart2Obj.txBuffer = buffer;
268 uart2Obj.txSize = size;
269 uart2Obj.txProcessedSize = 0;
270 uart2Obj.txBusyStatus = true;
271 status = true;
272
273 size_t txProcessedSize = uart2Obj.txProcessedSize;
274 size_t txSize = uart2Obj.txSize;
275
276 /* Initiate the transfer by writing as many bytes as we can */
277 while(((U2STA & _U2STA_UTXBF_MASK) == 0U) && (txSize > txProcessedSize) )
278 {
279 if (( U2MODE & (_U2MODE_PDSEL0_MASK | _U2MODE_PDSEL1_MASK)) == (_U2MODE_PDSEL0_MASK | _U2MODE_PDSEL1_MASK))
280 {
281 /* 9-bit mode */
282 U2TXREG = ((uint16_t*)uart2Obj.txBuffer)[txProcessedSize];
283 txProcessedSize++;
284 }
285 else
286 {
287 /* 8-bit mode */
288 U2TXREG = ((uint8_t*)uart2Obj.txBuffer)[txProcessedSize];
289 txProcessedSize++;
290 }
291 }
292
293 uart2Obj.txProcessedSize = txProcessedSize;
294
295 IEC1SET = _IEC1_U2TXIE_MASK;
296 }
297 }
298
299 return status;

366static void __attribute__((used)) UART2_FAULT_InterruptHandler (void)
367{
368 /* Save the error to be reported later */
369 uart2Obj.errors = (U2STA & (_U2STA_OERR_MASK | _U2STA_FERR_MASK | _U2STA_PERR_MASK));
370
371 /* Disable the fault interrupt */
372 IEC1CLR = _IEC1_U2EIE_MASK;
373
374 /* Disable the receive interrupt */
375 IEC1CLR = _IEC1_U2RXIE_MASK;
376
377 /* Clear rx status */
378 uart2Obj.rxBusyStatus = false;
379
380 UART2_ErrorClear();
381
382 /* Client must call UARTx_ErrorGet() function to get the errors */
383 if( uart2Obj.rxCallback != NULL )
384 {
385 uintptr_t rxContext = uart2Obj.rxContext;
386
387 uart2Obj.rxCallback(rxContext);
388 }
389}

391static void __attribute__((used)) UART2_RX_InterruptHandler (void)
392{
393 if(uart2Obj.rxBusyStatus == true)
394 {
395 size_t rxSize = uart2Obj.rxSize;
396 size_t rxProcessedSize = uart2Obj.rxProcessedSize;
397
398 while((_U2STA_URXDA_MASK == (U2STA & _U2STA_URXDA_MASK)) && (rxSize > rxProcessedSize) )
399 {
400 if (( U2MODE & (_U2MODE_PDSEL0_MASK | _U2MODE_PDSEL1_MASK)) == (_U2MODE_PDSEL0_MASK | _U2MODE_PDSEL1_MASK))
401 {
402 /* 9-bit mode */
403 ((uint16_t*)uart2Obj.rxBuffer)[rxProcessedSize] = (uint16_t)(U2RXREG);
404 }
405 else
406 {
407 /* 8-bit mode */
408 ((uint8_t*)uart2Obj.rxBuffer)[rxProcessedSize] = (uint8_t)(U2RXREG);
409 }
410 rxProcessedSize++;
411 }
412
413 uart2Obj.rxProcessedSize = rxProcessedSize;
414
415 /* Clear UART2 RX Interrupt flag */
416 IFS1CLR = _IFS1_U2RXIF_MASK;
417
418 /* Check if the buffer is done */
419 if(uart2Obj.rxProcessedSize >= rxSize)
420 {
421 uart2Obj.rxBusyStatus = false;
422
423 /* Disable the fault interrupt */
424 IEC1CLR = _IEC1_U2EIE_MASK;
425
426 /* Disable the receive interrupt */
427 IEC1CLR = _IEC1_U2RXIE_MASK;
428
429
430 if(uart2Obj.rxCallback != NULL)
431 {
432 uintptr_t rxContext = uart2Obj.rxContext;
433
434 uart2Obj.rxCallback(rxContext);
435 }
436 }
437 }
438 else
439 {
440 /* Nothing to process */
441 }

494void __attribute__((used)) UART_2_InterruptHandler (void)
495{
496 /* As per 13_5_violation using this temp variable */
497 uint32_t temp = 0;
498
499 temp = IEC1;
500 /* Call Error handler if Error interrupt flag is set */
501 if (((IFS1 & _IFS1_U2EIF_MASK) != 0U) && ((temp & _IEC1_U2EIE_MASK) != 0U))
502 {
503 UART2_FAULT_InterruptHandler();
504 }
505 temp = IEC1;
506 /* Call RX handler if RX interrupt flag is set */
507 if (((IFS1 & _IFS1_U2RXIF_MASK) != 0U) && ((temp & _IEC1_U2RXIE_MASK) != 0U))
508 {
509 UART2_RX_InterruptHandler();
510 }
511 temp = IEC1;
512 /* Call TX handler if TX interrupt flag is set */
513 if (((IFS1 & _IFS1_U2TXIF_MASK) != 0U) && ((temp & _IEC1_U2TXIE_MASK) != 0U))
514 {
515 UART2_TX_InterruptHandler();
516 }
517}
