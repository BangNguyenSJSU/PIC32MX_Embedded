69void static UART2_ErrorClear( void )
70{
71 UART_ERROR errors = UART_ERROR_NONE;
72 uint8_t dummyData = 0u;
73
74 errors = (UART_ERROR)(U2STA & (_U2STA_OERR_MASK | _U2STA_FERR_MASK | _U2STA_PERR_MASK));
75
76 if(errors != UART_ERROR_NONE)
77 {
78 /* If it's a overrun error then clear it to flush FIFO */
79 if((U2STA & _U2STA_OERR_MASK) != 0U)
80 {
81 U2STACLR = _U2STA_OERR_MASK;
82 }
83
84 /* Read existing error bytes from FIFO to clear parity and framing error flags */
85 while((U2STA & _U2STA_URXDA_MASK) != 0U)
86 {
87 dummyData = (uint8_t)U2RXREG;
88 }
89
90 /* Clear error interrupt flag */
91 IFS1CLR = _IFS1_U2EIF_MASK;
92
93 /* Clear up the receive interrupt flag so that RX interrupt is not
94 * triggered for error bytes */
95 IFS1CLR = _IFS1_U2RXIF_MASK;
96
97 }
98
99 // Ignore the warning
100 (void)dummyData;
101}

103void UART2_Initialize( void )
104{
105 /* Set up UxMODE bits */
106 /* STSEL = 0 */
107 /* PDSEL = 0 */
108
109 U2MODE = 0x8;
110
111 /* Enable UART2 Receiver and Transmitter */
112 U2STASET = (_U2STA_UTXEN_MASK | _U2STA_URXEN_MASK | _U2STA_UTXISEL1_MASK );
113
114 /* BAUD Rate register Setup */
115 U2BRG = 520;
116
117 /* Disable Interrupts */
118 IEC1CLR = _IEC1_U2EIE_MASK;
119
120 IEC1CLR = _IEC1_U2RXIE_MASK;
121
122 IEC1CLR = _IEC1_U2TXIE_MASK;
123
124 /* Initialize instance object */
125 uart2Obj.rdCallback = NULL;
126 uart2Obj.rdInIndex = 0;
127 uart2Obj.rdOutIndex = 0;
128 uart2Obj.isRdNotificationEnabled = false;
129 uart2Obj.isRdNotifyPersistently = false;
130 uart2Obj.rdThreshold = 0;
131
132 uart2Obj.wrCallback = NULL;
133 uart2Obj.wrInIndex = 0;
134 uart2Obj.wrOutIndex = 0;
135 uart2Obj.isWrNotificationEnabled = false;
136 uart2Obj.isWrNotifyPersistently = false;
137 uart2Obj.wrThreshold = 0;
138
139 uart2Obj.errors = UART_ERROR_NONE;
140
141 if (UART2_IS_9BIT_MODE_ENABLED())
142 {
143 uart2Obj.rdBufferSize = UART2_READ_BUFFER_SIZE_9BIT;
144 uart2Obj.wrBufferSize = UART2_WRITE_BUFFER_SIZE_9BIT;
145 }
146 else
147 {
148 uart2Obj.rdBufferSize = UART2_READ_BUFFER_SIZE;
149 uart2Obj.wrBufferSize = UART2_WRITE_BUFFER_SIZE;
150 }
151
152
153 /* Turn ON UART2 */
154 U2MODESET = _U2MODE_ON_MASK;
155
156 /* Enable UART2_FAULT Interrupt */
157 IEC1SET = _IEC1_U2EIE_MASK;
158
159 /* Enable UART2_RX Interrupt */
160 IEC1SET = _IEC1_U2RXIE_MASK;

163bool UART2_SerialSetup( UART_SERIAL_SETUP *setup, uint32_t srcClkFreq )
164{
165 bool status = false;
166 uint32_t baud;
167 uint32_t status_ctrl;
168 uint32_t uxbrg = 0;
169
170 if (setup != NULL)
171 {
172 baud = setup->baudRate;
173
174 if ((baud == 0U) || ((setup->dataWidth == UART_DATA_9_BIT) && (setup->parity != UART_PARITY_NONE)))
175 {
176 return status;
177 }
178
179 if(srcClkFreq == 0U)
180 {
181 srcClkFreq = UART2_FrequencyGet();
182 }
183
184 /* Calculate BRG value */
185 uxbrg = (((srcClkFreq >> 2) + (baud >> 1)) / baud);
186
187 /* Check if the baud value can be set with low baud settings */
188 if (uxbrg < 1U)
189 {
190 return status;
191 }
192
193 uxbrg -= 1U;
194
195 if (uxbrg > UINT16_MAX)
196 {
197 return status;
198 }
199
200 /* Turn OFF UART2. Save UTXEN, URXEN and UTXBRK bits as these are cleared upon disabling UART */
201
202 status_ctrl = U2STA & (_U2STA_UTXEN_MASK | _U2STA_URXEN_MASK | _U2STA_UTXBRK_MASK);
203
204 U2MODECLR = _U2MODE_ON_MASK;
205
206 if(setup->dataWidth == UART_DATA_9_BIT)
207 {
208 /* Configure UART2 mode */
209 U2MODE = (U2MODE & (~_U2MODE_PDSEL_MASK)) | setup->dataWidth;
210 }
211 else
212 {
213 /* Configure UART2 mode */
214 U2MODE = (U2MODE & (~_U2MODE_PDSEL_MASK)) | setup->parity;
215 }
216
217 /* Configure UART2 mode */
218 U2MODE = (U2MODE & (~_U2MODE_STSEL_MASK)) | setup->stopBits;
219
220 /* Configure UART2 Baud Rate */
221 U2BRG = uxbrg;
222
223 if (UART2_IS_9BIT_MODE_ENABLED())
224 {
225 uart2Obj.rdBufferSize = UART2_READ_BUFFER_SIZE_9BIT;
226 uart2Obj.wrBufferSize = UART2_WRITE_BUFFER_SIZE_9BIT;
227 }
228 else
229 {
230 uart2Obj.rdBufferSize = UART2_READ_BUFFER_SIZE;
231 uart2Obj.wrBufferSize = UART2_WRITE_BUFFER_SIZE;
232 }
233
234 U2MODESET = _U2MODE_ON_MASK;
235
236 /* Restore UTXEN, URXEN and UTXBRK bits. */
237 U2STASET = status_ctrl;
238
239 status = true;
240 }
241
242 return status;
243}

245/* This routine is only called from ISR. Hence do not disable/enable USART interrupts. */
246static inline bool UART2_RxPushByte(uint16_t rdByte)
247{
248 uint32_t tempInIndex;
249 bool isSuccess = false;
250 uint32_t rdInIdx;
251
252 tempInIndex = uart2Obj.rdInIndex + 1U;
253
254 if (tempInIndex >= uart2Obj.rdBufferSize)
255 {
256 tempInIndex = 0U;
257 }
258
259 if (tempInIndex == uart2Obj.rdOutIndex)
260 {
261 /* Queue is full - Report it to the application. Application gets a chance to free up space by reading data out from the RX ring buffer */
262 if(uart2Obj.rdCallback != NULL)
263 {
264 uintptr_t rdContext = uart2Obj.rdContext;
265
266 uart2Obj.rdCallback(UART_EVENT_READ_BUFFER_FULL, rdContext);
267
268 /* Read the indices again in case application has freed up space in RX ring buffer */
269 tempInIndex = uart2Obj.rdInIndex + 1U;
270
271 if (tempInIndex >= uart2Obj.rdBufferSize)
272 {
273 tempInIndex = 0U;
274 }
275 }
276 }
277
278 /* Attempt to push the data into the ring buffer */
279 if (tempInIndex != uart2Obj.rdOutIndex)
280 {
281 uint32_t rdInIndex = uart2Obj.rdInIndex;
282
283 if (UART2_IS_9BIT_MODE_ENABLED())
284 {
285 rdInIdx = uart2Obj.rdInIndex << 1U;
286 UART2_ReadBuffer[rdInIdx] = (uint8_t)rdByte;
287 UART2_ReadBuffer[rdInIdx + 1U] = (uint8_t)(rdByte >> 8U);
288 }
289 else
290 {
291 UART2_ReadBuffer[rdInIndex] = (uint8_t)rdByte;
292 }
293
294 uart2Obj.rdInIndex = tempInIndex;
295
296 isSuccess = true;
297 }
298 else
299 {
300 /* Queue is full. Data will be lost. */
301 }
302
303 return isSuccess;
304}

307static void UART2_ReadNotificationSend(void)
308{
309 uint32_t nUnreadBytesAvailable;
310
311 if (uart2Obj.isRdNotificationEnabled == true)
312 {
313 nUnreadBytesAvailable = UART2_ReadCountGet();
314
315 if(uart2Obj.rdCallback != NULL)
316 {
317 uintptr_t rdContext = uart2Obj.rdContext;
318
319 if (uart2Obj.isRdNotifyPersistently == true)
320 {
321 if (nUnreadBytesAvailable >= uart2Obj.rdThreshold)
322 {
323 uart2Obj.rdCallback(UART_EVENT_READ_THRESHOLD_REACHED, rdContext);
324 }
325 }
326 else
327 {
328 if (nUnreadBytesAvailable == uart2Obj.rdThreshold)
329 {
330 uart2Obj.rdCallback(UART_EVENT_READ_THRESHOLD_REACHED, rdContext);
331 }
332 }
333 }

337size_t UART2_Read(uint8_t* pRdBuffer, const size_t size)
338{
339 size_t nBytesRead = 0;
340 uint32_t rdOutIndex = 0;
341 uint32_t rdInIndex = 0;
342 uint32_t rdOut16Idx;
343 uint32_t nBytesRead16Idx;
344
345 /* Take a snapshot of indices to avoid creation of critical section */
346 rdOutIndex = uart2Obj.rdOutIndex;
347 rdInIndex = uart2Obj.rdInIndex;
348
349 while (nBytesRead < size)
350 {
351 if (rdOutIndex != rdInIndex)
352 {
353 if (UART2_IS_9BIT_MODE_ENABLED())
354 {
355 rdOut16Idx = rdOutIndex << 1U;
356 nBytesRead16Idx = nBytesRead << 1U;
357
358 pRdBuffer[nBytesRead16Idx] = UART2_ReadBuffer[rdOut16Idx];
359 pRdBuffer[nBytesRead16Idx + 1U] = UART2_ReadBuffer[rdOut16Idx + 1U];
360 }
361 else
362 {
363 pRdBuffer[nBytesRead] = UART2_ReadBuffer[rdOutIndex];
364 }
365 nBytesRead++;
366 rdOutIndex++;
367
368 if (rdOutIndex >= uart2Obj.rdBufferSize)
369 {
370 rdOutIndex = 0U;
371 }
372 }
373 else
374 {
375 /* No more data available in the RX buffer */
376 break;
377 }
378 }
379
380 uart2Obj.rdOutIndex = rdOutIndex;
381
382 return nBytesRead;
383}

444static bool UART2_TxPullByte(uint16_t* pWrByte)
445{
446 bool isSuccess = false;
447 uint32_t wrOutIndex = uart2Obj.wrOutIndex;
448 uint32_t wrInIndex = uart2Obj.wrInIndex;
449 uint32_t wrOut16Idx;
450
451 if (wrOutIndex != wrInIndex)
452 {
453 if (UART2_IS_9BIT_MODE_ENABLED())
454 {
455 wrOut16Idx = wrOutIndex << 1U;
456 pWrByte[0] = UART2_WriteBuffer[wrOut16Idx];
457 pWrByte[1] = UART2_WriteBuffer[wrOut16Idx + 1U];
458 }
459 else
460 {
461 *pWrByte = UART2_WriteBuffer[wrOutIndex];
462 }
463 wrOutIndex++;
464
465 if (wrOutIndex >= uart2Obj.wrBufferSize)
466 {
467 wrOutIndex = 0U;
468 }
469
470 uart2Obj.wrOutIndex = wrOutIndex;
471
472 isSuccess = true;
473 }
474
475 return isSuccess;

578size_t UART2_Write(uint8_t* pWrBuffer, const size_t size )
579{
580 size_t nBytesWritten = 0;
581 uint16_t halfWordData = 0U;
582
583 while (nBytesWritten < size)
584 {
585 if (UART2_IS_9BIT_MODE_ENABLED())
586 {
587 halfWordData = pWrBuffer[(2U * nBytesWritten) + 1U];
588 halfWordData <<= 8U;
589 halfWordData |= pWrBuffer[(2U * nBytesWritten)];
590 if (UART2_TxPushByte(halfWordData) == true)
591 {
592 nBytesWritten++;
593 }
594 else
595 {
596 /* Queue is full, exit the loop */
597 break;
598 }
599 }
600 else
601 {
602 if (UART2_TxPushByte(pWrBuffer[nBytesWritten]) == true)
603 {
604 nBytesWritten++;
605 }
606 else
607 {
608 /* Queue is full, exit the loop */
609 break;
610 }
611 }
612
613 }
614
615 /* Check if any data is pending for transmission */
616 if (UART2_WritePendingBytesGet() > 0U)
617 {
618 /* Enable TX interrupt as data is pending for transmission */
619 UART2_TX_INT_ENABLE();
620 }
621
622 return nBytesWritten;
623}

718static void __attribute__((used)) UART2_RX_InterruptHandler (void)
719{
720 /* Keep reading until there is a character availabe in the RX FIFO */
721 while((U2STA & _U2STA_URXDA_MASK) == _U2STA_URXDA_MASK)
722 {
723 if (UART2_RxPushByte( (uint16_t )(U2RXREG) ) == true)
724 {
725 UART2_ReadNotificationSend();
726 }
727 else
728 {
729 /* UART RX buffer is full */
730 }
731 }
732
733 /* Clear UART2 RX Interrupt flag */
734 IFS1CLR = _IFS1_U2RXIF_MASK;
735}

737static void __attribute__((used)) UART2_TX_InterruptHandler (void)
738{
739 uint16_t wrByte;
740
741 /* Check if any data is pending for transmission */
742 if (UART2_WritePendingBytesGet() > 0U)
743 {
744 /* Keep writing to the TX FIFO as long as there is space */
745 while((U2STA & _U2STA_UTXBF_MASK) == 0U)
746 {
747 if (UART2_TxPullByte(&wrByte) == true)
748 {
749 if (UART2_IS_9BIT_MODE_ENABLED())
750 {
751 U2TXREG = wrByte;
752 }
753 else
754 {
755 U2TXREG = (uint8_t)wrByte;
756 }
757
758 /* Send notification */
759 UART2_WriteNotificationSend();
760 }
761 else
762 {
763 /* Nothing to transmit. Disable the data register empty interrupt. */
764 UART2_TX_INT_DISABLE();
765 break;
766 }
767 }
768
769 /* Clear UART2TX Interrupt flag */
770 IFS1CLR = _IFS1_U2TXIF_MASK;
771 }
772 else
773 {
774 /* Nothing to transmit. Disable the data register empty interrupt. */
775 UART2_TX_INT_DISABLE();
776
777 /* Clear UART2TX Interrupt flag */
778 IFS1CLR = _IFS1_U2TXIF_MASK;
779 }
780}

783{
784 uint32_t temp = IEC1;
785 /* Call Error handler if Error interrupt flag is set */
786 if (((IFS1 & _IFS1_U2EIF_MASK) != 0U) && ((temp & _IEC1_U2EIE_MASK) != 0U))
