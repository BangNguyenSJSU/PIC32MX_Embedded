85static void SwitchPress_Handler( CN_PIN cnPin, uintptr_t context)
86{
87 BaseType_t xHigherPriorityTaskWoken;
88
89 /* Unblock the task by releasing the semaphore. */

94
95 portEND_SWITCHING_ISR( xHigherPriorityTaskWoken );

133/******************************************************************************
134 Function:
135 void APP_TASK3_Tasks ( void )
136
137 Remarks:
138 This task blocks on switch press. It unblocks when switch is pressed/released, toggles LED and
139 again blocks on switch press event.
140 */
141
142void APP_TASK3_Tasks ( void )
143{
144 bool status = false;
145
146 switchPressSemaphore = xSemaphoreCreateBinary();
147
148 if (switchPressSemaphore != NULL)
149 {
150 status = true;
151 }
152
153 while (status == true)
154 {
155 /* Block until user presses the switch */
156 if( xSemaphoreTake( switchPressSemaphore, portMAX_DELAY ) == pdTRUE )
157 {
158 LED3_Toggle();
159 }
160 }
161}
