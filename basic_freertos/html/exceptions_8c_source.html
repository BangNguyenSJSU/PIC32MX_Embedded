5 exceptions.c

7 Summary:
8 This file contains a function which overrides the default _weak_ exception
9 handler provided by the XC32 compiler.
10
11 Description:
12 This file redefines the default _weak_ exception handler with a more debug
13 friendly one. If an unexpected exception occurs the code will stop in a
14 while(1) loop. The debugger can be halted and two variables exception_code
15 and exception_address can be examined to determine the cause and address
16 where the exception occurred.

103/*******************************************************************************
104 Function:
105 void _general_exception_handler ( void )
106
107 Description:
108 A general exception is any non-interrupt exception which occurs during program
109 execution outside of bootstrap code.
110
111 Remarks:
112 Refer to the XC32 User's Guide for additional information.
113 */
114
115void __attribute__((noreturn, weak)) _general_exception_handler ( void )
116{
117 /* Mask off the ExcCode Field from the Cause Register
118 Refer to the MIPs Software User's manual */
119 exception_code = ((_CP0_GET_CAUSE() & 0x0000007CU) >> 2U);
120 exception_address = _CP0_GET_EPC();
121
122 while (true)
123 {
124 #if defined(__DEBUG) || defined(__DEBUG_D) && defined(__XC32)
125 __builtin_software_breakpoint();
126 #endif
127 }
128}

130/*******************************************************************************
131 Function:
132 void _bootstrap_exception_handler ( void )
133
134 Description:
135 A bootstrap exception is any exception which occurs while bootstrap code is
136 running (STATUS.BEV bit is 1).
137
138 Remarks:
139 Refer to the XC32 User's Guide for additional information.
140 */
141
142void __attribute__((noreturn, weak)) _bootstrap_exception_handler(void)
143{

148
