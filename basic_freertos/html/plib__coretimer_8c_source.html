49void CORETIMER_Initialize( void )
50{
51
52 // Clear Core Timer
53 _CP0_SET_COUNT(0);
54 _CP0_SET_COMPARE(compareValue);
55
56 // Enable Timer by clearing Disable Count (DC) bit
57 _CP0_SET_CAUSE(_CP0_GET_CAUSE() & (~_CP0_CAUSE_DC_MASK));
58}

60void CORETIMER_Start( void )
61{
62 // Disable Timer by setting Disable Count (DC) bit
63 _CP0_SET_CAUSE(_CP0_GET_CAUSE() | _CP0_CAUSE_DC_MASK);
64
65 // Clear Compare Timer Interrupt Flag
66 IFS0CLR = 0x1;
67
68 // Clear Core Timer
69 _CP0_SET_COUNT(0);
70
71 _CP0_SET_COMPARE(compareValue);
72
73 // Enable Timer by clearing Disable Count (DC) bit
74 _CP0_SET_CAUSE(_CP0_GET_CAUSE() & (~_CP0_CAUSE_DC_MASK));
75
76}

82}
