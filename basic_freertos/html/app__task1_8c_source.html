143/******************************************************************************
144 Function:
145 void APP_TASK1_Tasks ( void )
146
147 Remarks:
148 This task periodically runs every 1 second and toggles LED1
149 */
150
151void APP_TASK1_Tasks(void) {
152
153 switchPressToLCDSemaphore = xSemaphoreCreateBinary();
154
155
156 while (1) {
157 LED1_Toggle();
158 LCD_PRINT_STRING("RTOS !!", 0, 0, 3);
159 if(xSemaphoreTakeFromISR( switchPressToLCDSemaphore, pdFALSE ) == pdTRUE)
160 {
161 LCD_PRINT_STRING("SW_ON !!",5, 2, 3);
162 LED2_Toggle();
163 }
164 else
165 {
166 LCD_PRINT_STRING("SW_OFF !!",5, 2, 3);
167 }
168
169
170
171 /* Blink LED at every 1000 ms. Meanwhile allow other ready tasks to run */
172 vTaskDelay(1000 / portTICK_PERIOD_MS);
173 }
174}
