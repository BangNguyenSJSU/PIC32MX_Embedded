69static void DMAC_ChannelSetAddresses(DMAC_CHANNEL channel, const void *srcAddr, const void *destAddr)
70{
71 const uint32_t *xsrcAddr = (const uint32_t *)srcAddr;
72 const uint32_t *xdestAddr = (const uint32_t *)destAddr;
73
74 uint32_t sourceAddress = (uint32_t)xsrcAddr;
75 uint32_t destAddress = (uint32_t)xdestAddr;
76 volatile uint32_t * regs;
77
78 /* Set the source address */
79 /* Check if the address lies in the KSEG2 for MZ devices */
80 if ((sourceAddress >> 29) == 0x6U)
81 {
82 if ((sourceAddress >> 28)== 0xcU)
83 {
84 // EBI Address translation
85 sourceAddress = ((sourceAddress | 0x20000000U) & 0x2FFFFFFFU);
86 }
87 else if((sourceAddress >> 28)== 0xDU)
88 {
89 //SQI Address translation
90 sourceAddress = ((sourceAddress | 0x30000000U) & 0x3FFFFFFFU);
91 }
92 else
93 {
94 /* Do Nothing */
95 }
96 }
97 else if ((sourceAddress >> 29) == 0x7U)
98 {
99 if ((sourceAddress >> 28)== 0xEU)
100 {
101 // EBI Address translation
102 sourceAddress = ((sourceAddress | 0x20000000U) & 0x2FFFFFFFU);
103 }
104 else if ((sourceAddress >> 28)== 0xFU)
105 {
106 // SQI Address translation
107 sourceAddress = ((sourceAddress | 0x30000000U) & 0x3FFFFFFFU);
108 }
109 else
110 {
111 /* Do Nothing */
112 }
113 }
114 else
115 {
116 /* For KSEG0 and KSEG1, The translation is done by KVA_TO_PA */
117 sourceAddress = ConvertToPhysicalAddress(sourceAddress);
118 }
119
120 /* Set the source address, DCHxSSA */
121 regs = (volatile uint32_t *)(_DMAC_BASE_ADDRESS + 0x60U + (channel * 0xC0U) + 0x30U);
122 *(volatile uint32_t *)(regs) = sourceAddress;
123
124 /* Set the destination address */
125 /* Check if the address lies in the KSEG2 for MZ devices */
126 if ((destAddress >> 29) == 0x6U)
127 {
128 // EBI Address translation
129 if ((destAddress >> 28)== 0xcU)
130 {
131 destAddress = ((destAddress | 0x20000000U) & 0x2FFFFFFFU);
132 }
133 //SQI Address translation
134 else if ((destAddress >> 28)== 0xdU)
135 {
136 destAddress = ((destAddress | 0x30000000U) & 0x3FFFFFFFU);
137 }
138 else
139 {
140 /* Do Nothing */
141 }
142 }
143 else if ((destAddress >> 29) == 0x7U)
144 { /* Check if the address lies in the KSEG3 for MZ devices */
145 // EBI Address translation
146 if ((destAddress >> 28)== 0xeU)
147 {
148 destAddress = ((destAddress | 0x20000000U) & 0x2FFFFFFFU);
149 }
150 //SQI Address translation
151 else if ((destAddress >> 28)== 0xfU)
152 {
153 destAddress = ((destAddress | 0x30000000U) & 0x3FFFFFFFU);
154 }
155 else
156 {
157 /* Do Nothing */
158 }
159 }
160 else
161 {
162 /* For KSEG0 and KSEG1, The translation is done by KVA_TO_PA */
163 destAddress = ConvertToPhysicalAddress(destAddress);
164 }
165
166 /* Set destination address, DCHxDSA */
167 regs = (volatile uint32_t *)(_DMAC_BASE_ADDRESS + 0x60U + (channel * 0xC0U) + 0x40U);
168 *(volatile uint32_t *)(regs) = destAddress;
169}

172/* Function:
173 static uint32_t DMAC_BitReverse( uint32_t num, uint32_t bits)
174
175 Summary:
176 Reverses the bits in the given number
177
178 Description:
179 Reverses the bits in the given number based on the size of the number.
180 Example:
181 number = 10110011
182 reverse = 11001101
183
184 Parameters:
185 num - Number to be reversed
186 bits - size of the number (8, 16, 32)
187
188 Returns:
189 reversed number
190*/
191static uint32_t DMAC_BitReverse( uint32_t num, uint32_t bits)
192{
193 uint32_t out = 0;
194 uint32_t i;
195
196 for( i = 0U; i < bits; i++ )
197 {
198 out <<= 1;
199
200 if(( num & 1U ) != 0U)
201 {
202 out |= 1U;
203 }
204
205 num >>= 1;
206 }
207
208 return out;
209}

217void DMAC_Initialize(void)
218{
219 uint8_t chanIndex;
220 volatile DMAC_CHANNEL_OBJECT *chanObj;
221
222 /* Enable the DMA module */
223 DMACONSET = _DMACON_ON_MASK;
224
225 /* Initialize the available channel objects */
226 chanObj = &gDMAChannelObj[0];
227
228 for(chanIndex = 0U; chanIndex < 8U; chanIndex++)
229 {
230 chanObj->inUse = false;
231 chanObj->pEventCallBack = NULL;
232 chanObj->hClientArg = 0;
233 chanObj->errorInfo = DMAC_ERROR_NONE;
234 chanObj++; /* linked list 'next' */
235 }
236
237 /* DMACON register */
238 /* ON = 1 */
239 DMACON = 0x8000;
240
241 /* DMA channel-level control registers. They will have additional settings made when starting a transfer. */
242 /* DMA channel 0 configuration */
243 /* CHPRI = 3, CHAEN= 1, CHCHN= 0, CHCHNS= 0x0, CHAED= 0 */
244 DCH0CON = 0x13;
245 /* CHSIRQ = 41, SIRQEN = 1 */
246 DCH0ECON = 0x2910;
247 /* CHBCIE = 1, CHTAIE=1, CHERIE=1, CHSHIE= 1, CHDHIE= 0 */
248 DCH0INT = 0x4b0000;
249
250 /* Enable DMA channel interrupts */
251 IEC1SET = 0U | 0x10000U ;
252}

261bool DMAC_ChannelTransfer(DMAC_CHANNEL channel, const void *srcAddr, size_t srcSize, const void *destAddr, size_t destSize, size_t cellSize)
262{
263 bool returnStatus = false;
264 volatile uint32_t *regs;
265 uint32_t DCHxINT_Flags;
266
267 regs = ((volatile uint32_t *)(_DMAC_BASE_ADDRESS + 0x60U + (channel * 0xC0U) + 0x20U));
268 DCHxINT_Flags = *(volatile uint32_t *)(regs) & (_DCH0INT_CHERIF_MASK | _DCH0INT_CHTAIF_MASK | _DCH0INT_CHBCIF_MASK);
269
270 if((gDMAChannelObj[channel].inUse == false) || (DCHxINT_Flags != 0U))
271 {
272 /* Clear all the interrupt flags */
273 regs = (volatile uint32_t *)(_DMAC_BASE_ADDRESS + 0x60U + (channel * 0xC0U) + 0x20U) + 1U;
274 *(volatile uint32_t *)(regs) = (_DCH0INT_CHSHIF_MASK |_DCH0INT_CHDHIF_MASK | _DCH0INT_CHBCIF_MASK | _DCH0INT_CHTAIF_MASK| _DCH0INT_CHERIF_MASK);
275
276 gDMAChannelObj[channel].inUse = true;
277 returnStatus = true;
278
279 /* Set the source / destination addresses, DCHxSSA and DCHxDSA */
280 DMAC_ChannelSetAddresses(channel, srcAddr, destAddr);
281
282 /* Set the source size, DCHxSSIZ */
283 regs = (volatile uint32_t *)(_DMAC_BASE_ADDRESS + 0x60U + (channel * 0xC0U) + 0x50U);
284 *(volatile uint32_t *)(regs) = srcSize;
285
286 /* Set the destination size, DCHxDSIZ */
287 regs = (volatile uint32_t *)(_DMAC_BASE_ADDRESS + 0x60U + (channel * 0xC0U) + 0x60U);
288 *(volatile uint32_t *)(regs) = destSize;
289
290 /* Set the cell size, DCHxCSIZ */
291 regs = (volatile uint32_t *)(_DMAC_BASE_ADDRESS + 0x60U + (channel * 0xC0U) + 0x90U);
292 *(volatile uint32_t *)(regs) = cellSize;
293
294 /* Enable the channel */
295 /* CHEN = 1 */
296 regs = (volatile uint32_t *)(_DMAC_BASE_ADDRESS + 0x60U + (channel * 0xC0U) + 0x0U)+2U;
297 *(volatile uint32_t *)(regs) = _DCH0CON_CHEN_MASK;
298
299 /* Check Channel Start IRQ Enable bit - SIRQEN */
300 regs = (volatile uint32_t *)(_DMAC_BASE_ADDRESS + 0x60U + (channel * 0xC0U) + 0x10U);
301
302 /* Initiate transfer if user did not set up channel for interrupt-initiated transfer. */
303 if((*(volatile uint32_t *)(regs) & _DCH1ECON_SIRQEN_MASK) == 0U)
304 {
305 /* CFORCE = 1 */
306 regs = (volatile uint32_t *)(_DMAC_BASE_ADDRESS + 0x60U + (channel * 0xC0U) + 0x10U)+2U;
307 *(volatile uint32_t *)(regs) = _DCH0ECON_CFORCE_MASK;
308 }
309 }
310
311 return returnStatus;
312}

400DMAC_TRANSFER_EVENT DMAC_ChannelTransferStatusGet(DMAC_CHANNEL channel)
401{
402 uint32_t DCHxINT_Flags;
403
404 DMAC_TRANSFER_EVENT dmaEvent = DMAC_TRANSFER_EVENT_NONE;
405
406 DCHxINT_Flags = *(volatile uint32_t *)(_DMAC_BASE_ADDRESS + 0x60U + (channel * 0xC0U) + 0x20U);
407
408 if ((DCHxINT_Flags & (_DCH0INT_CHSHIF_MASK | _DCH0INT_CHDHIF_MASK)) != 0U) /* Dest Half-full or Src Half-empty flags */
409 {
410 dmaEvent = DMAC_TRANSFER_EVENT_HALF_COMPLETE;
411 }
412 if ((DCHxINT_Flags & (_DCH0INT_CHTAIF_MASK | _DCH0INT_CHERIF_MASK)) != 0U) /* Abort or Error flags*/
413 {
414 dmaEvent = DMAC_TRANSFER_EVENT_ERROR;
415 }
416 if ((DCHxINT_Flags & _DCH0INT_CHBCIF_MASK) != 0U) /* Block Transfer complete flag */
417 {
418 dmaEvent = DMAC_TRANSFER_EVENT_COMPLETE;
419 }
420
421 return dmaEvent;
422}

424void DMAC_ChannelCRCSetup( DMAC_CHANNEL channel, DMAC_CRC_SETUP CRCSetup )
425{
426 uint32_t mask = 0;
427
428 gCRCSetup.append_mode = CRCSetup.append_mode;
429 gCRCSetup.reverse_crc_input = CRCSetup.reverse_crc_input;
430 gCRCSetup.polynomial_length = CRCSetup.polynomial_length;
431 gCRCSetup.polynomial = CRCSetup.polynomial;
432 gCRCSetup.non_direct_seed = CRCSetup.non_direct_seed;
433 gCRCSetup.final_xor_value = CRCSetup.final_xor_value;
434 gCRCSetup.reverse_crc_output = CRCSetup.reverse_crc_output;
435
436 if (gCRCSetup.append_mode == true)
437 {
438 mask |= _DCRCCON_CRCAPP_MASK;
439 }
440
441 if (gCRCSetup.reverse_crc_input == true)
442 {
443 mask |= _DCRCCON_BITO_MASK;
444 }
445
446 mask |= (channel | _DCRCCON_CRCEN_MASK | (((uint32_t)gCRCSetup.polynomial_length - 1U) << _DCRCCON_PLEN_POSITION));
447
448 /* Setup the DMA CRCCON register */
449 DCRCCON = mask;
450
451 /* Store the polynomial value */
452 DCRCXOR = gCRCSetup.polynomial;
453
454 /* Store the Initial seed value */
455 DCRCDATA = gCRCSetup.non_direct_seed;
456}

463uint32_t DMAC_CRCRead( void )
464{
465 uint32_t crc = 0;
466
467 /* Read the generated CRC value.
468 * Once read DMAC_CRCEnable() has to be called again before DMA Transfer for new CRC.
469 */
470 crc = DCRCDATA;
471
472 /* Reverse the final crc value */
473 if (gCRCSetup.reverse_crc_output == true)
474 {
475 crc = DMAC_BitReverse(crc, gCRCSetup.polynomial_length);
476 }
477
478 crc ^= gCRCSetup.final_xor_value;
479
480 return crc;
481}

483void __attribute__((used)) DMA_0_InterruptHandler(void)
484{
485 volatile DMAC_CHANNEL_OBJECT *chanObj;
486 DMAC_TRANSFER_EVENT dmaEvent = DMAC_TRANSFER_EVENT_NONE;
487 uint32_t VarRead = 0;
488
489 /* Find out the channel object */
490 chanObj = &gDMAChannelObj[0];
491
492 /* Check whether the active DMA channel event has occurred */
493 VarRead = DCH0INTbits.CHDHIF;
494 if((DCH0INTbits.CHSHIF == 1U) || ( VarRead == 1U))/* irq due to half complete */
495 {
496 /* Do not clear the flag here, it should be cleared with block transfer complete flag*/
497
498 /* Update error and event */
499 chanObj->errorInfo = DMAC_ERROR_NONE;
500 dmaEvent = DMAC_TRANSFER_EVENT_HALF_COMPLETE;
501 /* Since transfer is only half done yet, do not make inUse flag false */
502 }
503 if(DCH0INTbits.CHTAIF == 1U) /* irq due to transfer abort */
504 {
505 /* Channel is by default disabled on Transfer Abortion */
506 /* Clear the Abort transfer complete flag */
507 DCH0INTCLR = _DCH0INT_CHTAIF_MASK;
508
509 /* Update error and event */
510 chanObj->errorInfo = DMAC_ERROR_NONE;
511 dmaEvent = DMAC_TRANSFER_EVENT_ERROR;
512 chanObj->inUse = false;
513 }
514 if(DCH0INTbits.CHBCIF == 1U) /* irq due to transfer complete */
515 {
516 /* Channel is by default disabled on completion of a block transfer */
517 /* Clear the Block transfer complete, half empty and half full interrupt flag */
518 DCH0INTCLR = _DCH0INT_CHBCIF_MASK | _DCH0INT_CHSHIF_MASK | _DCH0INT_CHDHIF_MASK;
519
520 /* Update error and event */
521 chanObj->errorInfo = DMAC_ERROR_NONE;
522 dmaEvent = DMAC_TRANSFER_EVENT_COMPLETE;
523 chanObj->inUse = false;
524 }
525 if(DCH0INTbits.CHERIF == 1U) /* irq due to address error */
526 {
527 /* Clear the address error flag */
528 DCH0INTCLR = _DCH0INT_CHERIF_MASK;
529
530 /* Update error and event */
531 chanObj->errorInfo = DMAC_ERROR_ADDRESS_ERROR;
532 dmaEvent = DMAC_TRANSFER_EVENT_ERROR;
533 chanObj->inUse = false;
534 }
535
536 /* Clear the interrupt flag and call event handler */
537 IFS1CLR = 0x10000;
538
539 if((chanObj->pEventCallBack != NULL) && (dmaEvent != DMAC_TRANSFER_EVENT_NONE))
540 {
541 uintptr_t hClientArg = chanObj->hClientArg;
542
543 chanObj->pEventCallBack(dmaEvent, hClientArg);
544 }
