57void TMR1_Initialize(void)
58{
59 /* Disable Timer */
60 T1CONCLR = _T1CON_ON_MASK;
61
62 /*
63 SIDL = 0
64 TWDIS = 0
65 TGATE = 0
66 TCKPS = 0
67 TSYNC = 0
68 TCS = 0
69 */
70 T1CONSET = 0x0;
71
72 /* Clear counter */
73 TMR1 = 0x0;
74
75 /*Set period */
76 PR1 = 11999;
77
78 /* Setup TMR1 Interrupt */
79 TMR1_InterruptEnable(); /* Enable interrupt on the way out */
80}

117void __attribute__((used)) TIMER_1_InterruptHandler (void)
118{
119 uint32_t status = IFS0bits.T1IF;
120 IFS0CLR = _IFS0_T1IF_MASK;
121
122 if((tmr1Obj.callback_fn != NULL))
123 {
124 uintptr_t context = tmr1Obj.context;
125 tmr1Obj.callback_fn(status, context);
126 }
127}
