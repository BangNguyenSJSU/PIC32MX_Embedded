196void
197_begin (uint8_t cols, uint8_t lines, uint8_t charsize)
198{
199 if (lines > 1)
200 {
201 _displayfunction |= LCD_2LINE;
202 }
203 _numlines = lines;
204
205 // for some 1 line displays you can select a 10 pixel high font
206 if ((charsize != 0) && (lines == 1))
207 {
208 _displayfunction |= LCD_5x10DOTS;
209 }
210
212 // SEE PAGE 45/46 FOR INITIALIZATION SPECIFICATION!
213 // according to datasheet, we need at least 40ms after power rises above 2.7V
214 // before sending commands. MCU can turnON at 4.5V (faster than LCD) so we'll wait 50
216 delay_coreTimer_ms (40);
217
218 // Now we pull both RS and R/W low to begin commands
219 expanderWrite (_backlightval); // reset expande_rand turn backlightOff (Bit 8 =1)
220 delay_coreTimer_ms (10); // 1000
221
223 // put the LCD into 4 bit mode
224 // this is according to the hitachi HD44780 datasheet
226 // Figure-24 - Page: 46
227
228 // we start in 8bit mode, try to set 4 bit mode
229 write4bits (0x30);
230 delay_coreTimer_ms (1); // wait min 1ms
231
232 // second try
233 write4bits (0x30);
234 delay_coreTimer_ms (1); // wait min 1ms
235
236 // third go!
237 write4bits (0x30);
238 delay_coreTimer_ms (1);
239
240 // finally, set to 4-bit interface
241 write4bits (0x20);
242
243 // set # lines, font size, etc.
244 _writeCommandBus (LCD_FUNCTIONSET | _displayfunction);
245
246 // turn the display on with no cursor or blinking default
247 _displaycontrol = LCD_DISPLAYON | LCD_CURSOROFF | LCD_BLINKOFF;
248 display ();
249
250 // clear it off
251 LCD_CLEAR_ALL ();
252
253 // Initialize to default text direction (for roman languages)
254 _displaymode = LCD_ENTRYLEFT | LCD_ENTRYSHIFTDECREMENT;
255
256 // set the entry mode
257 _writeCommandBus (LCD_ENTRYMODESET | _displaymode);
258
259 LCD_CURSOR_HOME ();
260}

272LCD_CLEAR_PARTICULAR (uint8_t rowStart, uint8_t colStart, uint8_t colCnt)

274 // Clear particular segment of a row

361LCD_PRINT_STRING (char *String_Input, size_t strSize, uint8_t rowStart, uint8_t colStart)
362{
363
364 size_t string_size;
365 // wrap-around case -->
366 if (rowStart == 2 && colStart < 21)
367 {
368 rowStart = 0;
369 colStart = colStart + 20;
370 }
371 else if (rowStart == 3 && colStart < 21)

378
379 // find the size of the string using strlen()

388
